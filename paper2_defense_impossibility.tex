\begin{abstract}
We characterize the geometric limits of prompt injection defense. We prove three results of increasing strength: (1)~any continuous, utility-preserving defense on a connected Hausdorff space must fix boundary points of the alignment deviation surface; (2)~when both defense and alignment function are Lipschitz, the defense cannot uniformly reduce alignment deviation far below threshold on a positive-measure band; (3)~under a transversality condition, the defense leaves a positive-measure region genuinely unsafe. We also establish parallel discrete results requiring no topology, and extend to multi-turn, stochastic, and pipeline settings. A representation-independent meta-theorem unifies all paths: utility preservation plus bounded regularity forces the defense's fixed-point set to spill beyond the safe region. The full theory is machine-verified in Lean~4 (39~files, $\sim$300 theorems, zero \texttt{sorry}, three standard axioms) and compared with empirical findings from three frontier LLMs~\cite{bhatt2025manifold}. Artifact: \url{https://github.com/mbhatt1/stuff/tree/main/ManifoldProofs}
\end{abstract}

\section{Introduction}
\label{sec:intro}

Can you build a wrapper around a language model that eliminates all prompt injection vulnerabilities?

The defense mechanisms proposed to date---input classifiers~\cite{alon2023detecting}, output filters~\cite{inan2023llama}, constitutional rewriting~\cite{bai2022constitutional}---share the same structure: a function $D\colon X \to X$ that preprocesses prompts, mapping unsafe inputs to safe equivalents while leaving safe inputs alone. We study the geometric limits of this mapping.

We prove three results of increasing strength. \textbf{Tier~1} (\Cref{thm:main}): if $D$ is continuous, utility-preserving, and the space is connected, $D$ must fix boundary points ($f(z)=\tau$, $D(z)=z$). \textbf{Tier~2} (\Cref{thm:eps-robust}): if also Lipschitz, $f(D(x)) \geq \tau - L(K+1)\delta$ for $x$ within distance $\delta$ of the fixed point. \textbf{Tier~3} (\Cref{thm:persistent}): under transversality, the defense leaves positive-measure volume genuinely unsafe. A representation-independent meta-theorem (\Cref{thm:meta}) unifies all paths: utility preservation plus any form of bounded regularity forces $\Fix(D) \supsetneq S_\tau$.

The novelty is not in the proof techniques---which are elementary---but in the problem formulation: identifying that standard defense design constraints induce a forced fixed-point structure at the safety boundary. All results are machine-verified in Lean~4 with zero \texttt{sorry} statements.

\section{Related Work}
\label{sec:related}

\paragraph{Adversarial robustness.} Prior work characterizes the existence of adversarial perturbations or provides certified robustness within bounded regions~\cite{szegedy2013intriguing,goodfellow2014explaining,cohen2019certified,singh2019abstract,katz2017reluplex}. Our result is a structural constraint on the defense mapping itself, independent of perturbation models or training.

\paragraph{Prompt injection.} Attack methods discover individual failures~\cite{zou2023universal,chao2024jailbreaking,mehrotra2024tree,samvelyan2024rainbow}. The Manifold of Failure framework~\cite{bhatt2025manifold} maps their topology. We prove these failures are structurally unavoidable under mild constraints.

\paragraph{Impossibility results.} No-free-lunch theorems~\cite{wolpert1997no} and robustness--accuracy tradeoffs~\cite{tsipras2018robustness} are distributional. Our results hold pointwise for any continuous defense on a fixed task.

\paragraph{Formal verification.} Lean and Coq have verified properties of neural networks~\cite{bagnall2019certifying} and optimization~\cite{bolte2024}. To our knowledge, ours is the first verified impossibility for prompt injection defense.

\section{Setup}
\label{sec:setup}

\begin{definition}
An \emph{alignment deviation function} is a continuous $f\colon X \to \R$. Given threshold $\tau$: $S_\tau = \{f < \tau\}$ (safe), $U_\tau = \{f > \tau\}$ (unsafe), $B_\tau = \{f = \tau\}$ (boundary). A \emph{defense} $D\colon X \to X$ is continuous, \emph{utility-preserving} if $D|_{S_\tau} = \id$, and \emph{complete} if $f(D(x)) < \tau$ for all $x$.
\end{definition}

\section{Tier~1: Boundary Fixation}
\label{sec:proof}

\begin{theorem}[Boundary Fixation]
\label{thm:main}
Let $X$ be a connected Hausdorff space, $f$ continuous with $S_\tau, U_\tau \neq \emptyset$, $D$ continuous with $D|_{S_\tau} = \id$. Then $\exists\, z$ with $f(z) = \tau$ and $D(z) = z$. Every $z \in \cl{S_\tau} \setminus S_\tau$ satisfies this; this set is nonempty.
\end{theorem}

\begin{proof}
(1)~$\Fix(D)$ is closed (Hausdorff $\Rightarrow$ diagonal closed). (2)~$S_\tau \subseteq \Fix(D)$; closed $\Rightarrow$ $\cl{S_\tau} \subseteq \Fix(D)$. (3)~$S_\tau$ is open, not closed (else clopen, contradicting connectedness). (4)~$\cl{S_\tau} \supsetneq S_\tau$; pick $z$ in the difference: $f(z) = \tau$. (5)~$z \in \Fix(D)$.
\end{proof}

\begin{corollary}
No continuous utility-preserving defense is complete.
\end{corollary}

Each hypothesis is necessary: disconnected $X$ allows complete defense; discontinuous $D$ allows complete defense; non-utility-preserving $D$ allows complete defense (counterexamples in \Cref{sec:counterexamples-app}).

\begin{remark}[Approximate utility preservation]
\label{rem:approx}
If $D$ satisfies $\dist(D(x), x) \leq \varepsilon$ for all $x \in S_\tau$ instead of strict identity, the fixed-point set $\Fix(D)$ may not contain $S_\tau$. However, the set $\{x : \dist(D(x), x) \leq \varepsilon\}$ is closed and contains $S_\tau$, so its closure-spillover argument still produces points near the boundary where $D$ moves inputs by at most $\varepsilon$. The impossibility degrades gracefully: boundary fixation becomes boundary \emph{near}-fixation with displacement $\leq \varepsilon$. A defense that ``jitters'' safe inputs escapes strict fixation but inherits an $\varepsilon$-approximate version.
\end{remark}

\section{Tier~2: $\varepsilon$-Robust Constraint}
\label{sec:eps-robust}

\begin{theorem}[$\varepsilon$-Robust Constraint]
\label{thm:eps-robust}
Under \Cref{thm:main}'s hypotheses, if $f$ is $L$-Lipschitz and $D$ is $K$-Lipschitz, the fixed point $z$ satisfies $f(D(x)) \geq \tau - L(K+1)\dist(x,z)$ for all $x$.
\end{theorem}

\begin{theorem}[Positive-Measure Band]
\label{thm:band-measure}
The $\varepsilon$-band $\{\tau - \varepsilon \leq f \leq \tau\}$ has positive measure and contains $\operatorname{ball}(c, \varepsilon/(4L))$ for $c$ with $f(c) = \tau - \varepsilon/2$.
\end{theorem}

\Cref{thm:eps-robust} constrains the \emph{depth} to which the defense can push near-boundary points below~$\tau$. It does not prove they remain unsafe---the defense may map them to $[\tau - L(K+1)\delta,\, \tau)$.

\section{Tier~3: Persistent Unsafe Region}
\label{sec:persistent}

\begin{definition}
The \emph{steep region} $\mathcal{S} = \{x : f(x) > \tau + L(K+1)\dist(x,z)\}$.
\end{definition}

\begin{theorem}[Persistent Unsafe Region]
\label{thm:persistent}
If $\mathcal{S} \neq \emptyset$: (1)~$\mathcal{S}$ is open; (2)~$\mu(\mathcal{S}) > 0$; (3)~$f(D(x)) > \tau$ for all $x \in \mathcal{S}$.
\end{theorem}

$\mathcal{S}$ is nonempty whenever the boundary slope exceeds $L(K+1)$---the surface rises faster than the defense can pull it down. Verified in Lean as \texttt{transversality\_from\_deriv}.

\section{Discrete-Continuous Bridge and Discrete Path}
\label{sec:bridge}

\begin{theorem}[Continuous Relaxation]
\label{thm:tietze}
Finite observations $g(p) < \tau < g(q)$ in a connected normal space admit continuous extensions for which Tiers~1--3 hold (Tietze/McShane--Whitney). This does not claim the results hold for the discrete token system directly, but for every continuous model consistent with the observations.
\end{theorem}

Independent of Tietze, we prove discrete results requiring no topology (\Cref{sec:discrete-app}): a discrete IVT on finite ordered sets, discrete defense fixation, and capacity exhaustion via pigeonhole.

\section{Extensions}
\label{sec:extensions}

\textbf{Multi-turn} (\Cref{thm:multi-turn-app}): boundary fixation recurs at every turn; the attacker accumulates information and can steer toward transversality. \textbf{Stochastic}: if $g(x) = \mathbb{E}[f(D(x))]$ is continuous and utility-preserving in expectation, $g(z) = \tau$ at boundary points. \textbf{Capacity parity}: utility preservation consumes $|S_\tau|$ capacity units, giving the attacker a permanent edge. \textbf{Pipelines}: Lipschitz constants multiply through tool calls; $n$-stage pipeline has effective constant $K^n$, exponentially degrading the defense.

\section{Meta-Theorem}
\label{sec:meta}

\begin{theorem}[Regularity Implies Spillover]
\label{thm:meta}
If $D(x) = x$ for all $x \in S$ and $\Fix(D)$ cannot equal $S$ (bounded regularity), then $\Fix(D) \supsetneq S$ and $\exists\, x \notin S$ with $D(x) = x$.
\end{theorem}

Each path proves regularity differently (continuous: $S$ not clopen; discrete: IVT + pigeonhole; stochastic: $g-f$ extends to closure). The meta-theorem is representation-independent.

\section{Engineering Prescription}
\label{sec:engineering}

(1)~Make the boundary shallow: set $\tau$ so boundary behavior is benign (GPT-5-Mini). (2)~Reduce $L$: smoother surfaces raise the Tier~3 threshold. (3)~Reduce dimension: defense cost $= N^d$. (4)~Monitor the boundary: crossings persist under fine-tuning. (5)~Multi-turn context doesn't help. (6)~Randomization fails in expectation. (7)~Break capacity parity via architectural separation.

\section{Limitations}
\label{sec:limitations}

Tier~1 applies only at the boundary ($f(z) = \tau$). Tier~2 constrains depth, not direction. Tier~3 requires transversality. The Tietze bridge applies to interpolants, not the discrete system itself; positive-measure bands may be empty of valid token sequences (the sharpest open gap). Raising $\tau$ fails for binary safety constraints. See \Cref{sec:limitations-app} for full discussion.

\section{Conclusion}

We establish geometric limits on prompt injection defense via a three-tier impossibility hierarchy, unified by a representation-independent meta-theorem. The impossibility is a consequence of combining utility preservation with bounded regularity---not an artifact of any particular proof technique. The practical prescription: make the boundary shallow, smooth, and low-dimensional. Machine-verified in Lean~4: 39 files, zero \texttt{sorry}.

\section{Counterexamples}
\label{sec:counterexamples-app}

\begin{counterexample}[Removing connectedness]
$X = \{0,1\}$ discrete, $f(0)=0$, $f(1)=1$, $\tau=0.5$. $D(0)=0$, $D(1)=0$: continuous, utility-preserving, complete.
\end{counterexample}

\begin{counterexample}[Removing continuity]
$X=[0,1]$, $f(x)=x$, $\tau=0.5$. $D(x)=x$ for $x<0.5$, $D(x)=0$ for $x \geq 0.5$.
\end{counterexample}

\begin{counterexample}[Removing utility preservation]
$D(x) = x_0$ for a fixed safe point.
\end{counterexample}

\section{Discrete Impossibility}
\label{sec:discrete-app}

\begin{theorem}[Discrete IVT]
$f\colon \{0,\ldots,n+1\} \to \R$ with $f(0) < \tau \leq f(n+1)$: $\exists\, i$ with $f(i) < \tau$ and $f(i+1) \geq \tau$.
\end{theorem}

\begin{theorem}[Discrete Defense Fixation]
If $D$ fixes safe vertices, the crossing has its safe endpoint fixed.
\end{theorem}

\begin{theorem}[Capacity Exhaustion]
$m$ configs, $< m$ defense behaviors $\Rightarrow$ pigeonhole collision.
\end{theorem}

\section{Multi-Turn Details}
\label{sec:multi-turn-app}

\begin{theorem}[Multi-Turn Impossibility]
\label{thm:multi-turn-app}
$\{f_t, D_t\}_{t=1}^T$ with each $f_t$ continuous, mixed regions, $D_t$ utility-preserving: $\forall\, t$, $\exists\, z_t$ with $f_t(z_t) = \tau$, $D_t(z_t) = z_t$.
\end{theorem}

The attacker's best score is monotone non-decreasing. It can steer toward transversality via binary search on its influence parameter.

\section{Vulnerability Landscape}
\label{sec:landscape-app}

\begin{theorem}[Basin Structure]
\label{thm:basin-app}
$U_\tau$ is open with positive measure (under any measure positive on nonempty open sets).
\end{theorem}

\begin{theorem}[Fragment Size]
$L$-Lipschitz $f$, $f(p) > \tau$: component diameter $\geq 2(f(p)-\tau)/L$.
\end{theorem}

\begin{theorem}[Perturbation Robustness]
$\operatorname{ball}(p, (f(p)-\tau)/L) \subseteq U_\tau$.
\end{theorem}

\begin{theorem}[Convergence]
Monotone operators on $[0,1]$ converge in $\leq \lfloor 1/\delta \rfloor$ steps.
\end{theorem}

\begin{theorem}[Transferability]
$\|f-g\|_\infty \leq \delta \Rightarrow \{f > \tau+\delta\} \subseteq \{g > \tau\}$.
\end{theorem}

\begin{theorem}[Authority Monotonicity]
Monotone in authority $\Rightarrow$ vulnerability upward-closed; threshold curve non-increasing if also monotone in indirection.
\end{theorem}

\begin{theorem}[Gradient Ascent]
Nonzero Fr\'{e}chet derivative $\Rightarrow$ $\exists$ ascent direction.
\end{theorem}

\section{Stability Under Fine-Tuning}
\label{sec:stability-app}

\begin{theorem}[Interior Stability]
\label{thm:interior-app}
$\|f-g\|_\infty \leq \varepsilon$: deep basin/safe points preserve classification.
\end{theorem}

\begin{theorem}[Crossing Preservation]
Transversal crossings persist under perturbation.
\end{theorem}

\begin{theorem}[Patching Is Nonlocal]
Eliminating a vulnerability at one point changes values elsewhere. (Lean construction.)
\end{theorem}

\section{Cost Asymmetry}

\begin{theorem}[Grid-Based Cost]
Grid defense: $N^d$ (exponential). Attack: $\leq 1/\delta$ (dimension-free).
\end{theorem}

\section{Pipeline Details}
\label{sec:pipeline-app}

\begin{theorem}[Pipeline Degradation]
\label{thm:pipeline-app}
$n$ stages compose to $(K_1 \cdots K_n)$-Lipschitz. Failure band scales as $L(K^n+1)\delta$.
\end{theorem}

\section{Extended Limitations}
\label{sec:limitations-app}

\paragraph{Jitter defense.} $D(x) \approx x$ breaks strict fixation but inherits $\varepsilon$-approximate version (\Cref{rem:approx}).

\paragraph{Binary exploits.} Theorem applies to smooth behavioral landscape, not discrete exploit chains.

\paragraph{Measure gap.} Positive-measure bands may be empty of valid tokens. Sharpest open gap.

\paragraph{Non-Hausdorff agents.} Geometric lock needs connectedness, not Hausdorff; discrete path needs no topology.

\paragraph{Monitoring blind spot.} Shallow boundary may allow tunneling into deep unsafe basins.

\paragraph{Binary safety constraints.} ``Raise $\tau$'' fails for cliff transitions.
